javascript#!/usr/bin/env node
const fs = require('fs');
const path = require('path');
const { execSync } = require('child_process');

// Find all copies of slider's package.json using find
try {
  const result = execSync(
    'find /Users/expo/workingdir/build/node_modules -name "package.json" -path "*/@react-native-community/slider/package.json" 2>/dev/null || true',
    { encoding: 'utf8' }
  ).trim();

  if (!result) {
    // Also try relative path from script location
    const appRoot = path.resolve(__dirname, '../..');
    const fallbackResult = execSync(
      `find ${appRoot}/node_modules -name "package.json" -path "*/@react-native-community/slider/package.json" 2>/dev/null || true`,
      { encoding: 'utf8' }
    ).trim();

    if (!fallbackResult) {
      console.log('[patch-slider] slider not found, nothing to patch');
      process.exit(0);
    }
  }

  const files = (result || '').split('\n').filter(Boolean);
  for (const filePath of files) {
    try {
      const realPath = fs.realpathSync(filePath);
      const pkg = JSON.parse(fs.readFileSync(realPath, 'utf8'));
      if (pkg.codegenConfig) {
        delete pkg.codegenConfig;
        fs.writeFileSync(realPath, JSON.stringify(pkg, null, 2));
        console.log(`[patch-slider] Removed codegenConfig from ${realPath}`);
      }
    } catch (e) {
      console.log(`[patch-slider] Could not patch ${filePath}: ${e.message}`);
    }
  }
} catch (e) {
  console.log('[patch-slider] find failed:', e.message);
}
